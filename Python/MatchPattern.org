  def describe(list):
      match list:
          case []:
              return "an empty list"
          case [single_item]:
              return f"a single-element list with {single_item}"
          case [first, second, *rest]:
              return f"a list with at least two elements: {first} and {second} and {rest}"
          case _:
              return "Default"

  print(describe([]))
  print(describe([1]))
  print(describe([1, 2, 3, 4, 5]))
  print(describe(3))
