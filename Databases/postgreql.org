* 查询
一共有多少连接数
#+begin_src sql
  SELECT count(*) FROM pg_stat_activity;
#+end_src

每个db有多少个连接数
#+begin_src sql
  SELECT datname, count(*) as connections
  FROM pg_stat_activity
  GROUP BY datname
  ORDER BY connections DESC;
#+end_src
