  from typing import Optional

  class ListNode:
      def __init__(self, val=0, next=None):
          self.val = val
          self.next = next

  class Solution:
      def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
          if head is None:
              return None
          slower, faster = head, head.next
          prev = None
          while faster:
              slower.next = prev
              prev = faster.next
              faster.next = slower
              slower = faster
              faster = prev
              prev = slower.next
          return slower
